55   implicit object AggregateWriter
56       extends OWrites[ResolvedCollectionCommand[Aggregate]] {
57     def writes(agg: ResolvedCollectionCommand[Aggregate]): JsObject = {
58       val fields = Seq[(String, JsValue)](
59         "aggregate" -> toJson(agg.collection),
60         "pipeline" -> JsArray(agg.command.pipeline.map(_.makePipe)),
61         "explain" -> toJson(agg.command.explain),
62         "allowDiskUse" -> toJson(agg.command.allowDiskUse)
63       )
64 
65       def optFields: List[Option[(String, JsValue)]] = List(
66         agg.command.cursor.map(c => "cursor" -> Json.obj(
67           "batchSize" -> c.batchSize
68         ))
69       )
70 
71       def options = {
72         if (agg.command.wireVersion < MongoWireVersion.V32) optFields.flatten
73         else (optFields ++ List(
74           Some(agg.command.bypassDocumentValidation).map(by =>
75             "bypassDocumentValidation" -> toJson(by)),
76           agg.command.readConcern.map(rc => "readConcern" -> toJson(rc.level))
77         )).flatten
78       }
79 
80       JsObject(fields ++ options)
81     }
82   }
83 
84   implicit object AggregationResultReader
85       extends DealingWithGenericCommandErrorsReader[AggregationResult] {
86 
87     def readResult(doc: JsObject): AggregationResult =
88       if (doc.value contains "stages") AggregationResult(List(doc), None) else {
89         (doc \ "result").validateOpt[List[JsObject]].flatMap {
90           case Some(docs) => JsSuccess(AggregationResult(docs, None))
91           case _ => for {
92             cursor <- (doc \ "cursor").validate[JsObject]
93             id <- (cursor \ "id").validate[Long]
94             ns <- (cursor \ "ns").validate[String]
95             firstBatch <- (cursor \ "firstBatch").validate[List[JsObject]]
96           } yield AggregationResult(firstBatch, Some(ResultCursor(id, ns)))
97         }.get
98       }
99   }
