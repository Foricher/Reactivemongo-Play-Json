154   class BSONDocumentFormat(toBSON: JsValue => JsResult[BSONValue], toJSON: BSONValue => JsValue) extends PartialFormat[BSONDocument] {
155     val partialReads: PartialFunction[JsValue, JsResult[BSONDocument]] = {
156       case obj @ JsObject(_) =>
157         try {
158           JsSuccess(bson(obj))
159         } catch {
160           case e: Throwable => JsError(e.getMessage)
161         }
162     }
163 
164     val partialWrites: PartialFunction[BSONValue, JsValue] = {
165       case doc: BSONDocument => json(doc)
166     }
167 
168     // UNSAFE - FOR INTERNAL USE
169     private[json] def bson(obj: JsObject): BSONDocument = BSONDocument(
170       obj.fields.map { tuple =>
171         tuple._1 -> (toBSON(tuple._2) match {
172           case JsSuccess(bson, _) => bson
173           case JsError(err)       => throw new JSONException(err.toString)
174         })
175       }
176     )
177 
178     // UNSAFE - FOR INTERNAL USE
179     private[json] def json(bson: BSONDocument): JsObject =
180       JsObject(bson.elements.map(elem => elem.name -> toJSON(elem.value)))

311   implicit object BSONRegexFormat extends PartialFormat[BSONRegex] {
312     val partialReads: PartialFunction[JsValue, JsResult[BSONRegex]] = {
313       case js: JsObject if js.values.size == 1 && js.fields.head._1 == "$regex" =>
314         js.fields.head._2.asOpt[String].
315           map(rx => JsSuccess(BSONRegex(rx, ""))).
316           getOrElse(JsError(__ \ "$regex", "string expected"))
317       case js: JsObject if js.value.size == 2 && js.value.exists(_._1 == "$regex") && js.value.exists(_._1 == "$options") =>
318         val rx = (js \ "$regex").asOpt[String]
319         val opts = (js \ "$options").asOpt[String]
320         (rx, opts) match {
321           case (Some(rx), Some(opts)) => JsSuccess(BSONRegex(rx, opts))
322           case (None, Some(_))        => JsError(__ \ "$regex", "string expected")
323           case (Some(_), None)        => JsError(__ \ "$options", "string expected")
324           case _                      => JsError(__ \ "$regex", "string expected") ++ JsError(__ \ "$options", "string expected")
325         }
326     }
327     val partialWrites: PartialFunction[BSONValue, JsValue] = {
328       case rx: BSONRegex =>
329         if (rx.flags.isEmpty)
330           Json.obj("$regex" -> rx.value)
331         else Json.obj("$regex" -> rx.value, "$options" -> rx.flags)
332     }

506   implicit class JsPathMongo(val jp: JsPath) extends AnyVal {
507     def writemongo[A](implicit writer: Writes[A]): OWrites[A] = {
508       OWrites[A] { (o: A) =>
509         val newPath = jp.path.flatMap {
510           case e: KeyPathNode     => Some(e.key)
511           case e: RecursiveSearch => Some(s"$$.${e.key}")
512           case e: IdxPathNode     => Some(s"${e.idx}")
513         }.mkString(".")
514 
515         val orig = writer.writes(o)
516         orig match {
517           case JsObject(e) =>
518             JsObject(e.flatMap {
519               case (k, v) => Seq(s"${newPath}.$k" -> v)
520             })
521           case e: JsValue => JsObject(Seq(newPath -> e))
522         }
523       }
524     }
525   }

528 object JSONSerializationPack extends reactivemongo.api.SerializationPack {
529   import reactivemongo.bson.buffer.{ ReadableBuffer, WritableBuffer }
530 
531   type Value = JsValue
532   type ElementProducer = (String, Json.JsValueWrapper)
533   type Document = JsObject
534   type Writer[A] = OWrites[A]
535   type Reader[A] = Reads[A]
536   type NarrowValueReader[A] = Reads[A]
537   private[reactivemongo]type WidenValueReader[A] = Reads[A]
538 
539   object IdentityReader extends Reader[Document] {
540     def reads(js: JsValue): JsResult[Document] = js match {
541       case o: JsObject => JsSuccess(o)
542       case v           => JsError(s"object is expected: $v")
543     }
544   }
545 
546   object IdentityWriter extends Writer[Document] {
547     def writes(document: Document): Document = document
548   }
549 
550   def serialize[A](a: A, writer: Writer[A]): Document = writer.writes(a)
551 
552   def deserialize[A](document: Document, reader: Reader[A]): A =
553     reader.reads(document) match {
554       case JsError(msg)    => sys.error(msg mkString ", ")
555       case JsSuccess(v, _) => v
556     }
557 
558   def writeToBuffer(buffer: WritableBuffer, document: Document): WritableBuffer = BSONFormats.toBSON(document) match {
559     case err @ JsError(_) => sys.error(s"fails to write the document: $document: ${Json stringify JsError.toJson(err)}")
560 
561     case JsSuccess(d @ BSONDocument(_), _) => {
562       BSONDocument.write(d, buffer)
563       buffer
564     }
565 
566     case JsSuccess(v, _) => sys.error(s"fails to write the document: $document; unexpected conversion $v")
567   }
568 
569   def readFromBuffer(buffer: ReadableBuffer): Document =
570     BSONFormats.toJSON(BSONDocument.read(buffer)).as[Document]
571 
572   def writer[A](f: A => Document): Writer[A] = new OWrites[A] {
573     def writes(input: A): Document = f(input)
574   }
575 
576   def isEmpty(document: Document): Boolean = document.values.isEmpty
577 
578   def widenReader[T](r: NarrowValueReader[T]): WidenValueReader[T] = r
579 
580   def readValue[A](value: Value, reader: WidenValueReader[A]): Try[A] =
581     reader.reads(value) match {
582       case err @ JsError(_) => Failure(new scala.RuntimeException(s"fails to reads the value: ${Json stringify value}; ${Json stringify JsError.toJson(err)}"))
583 
584       case JsSuccess(v, _)  => Success(v)
585     }
586 
587 }
