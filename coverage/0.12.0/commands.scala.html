56 private[commands] trait DealingWithGenericCommandErrorsReader[A]
57     extends Reads[A] {
58 
59   def readResult(doc: JsObject): A
60 
61   final def reads(json: JsValue): JsResult[A] = json match {
62     case doc: JsObject => {
63       if (!(doc \ "ok").asOpt[JsNumber].exists(_.value.toInt == 1)) {
64         JsError(new DefaultJSONCommandError(
65           code = (doc \ "code").asOpt[Int],
66           errmsg = (doc \ "errmsg").asOpt[String],
67           originalDocument = doc
68         ).getMessage())
69       } else JsSuccess(readResult(doc))
70     }
71 
72     case v => JsError(s"Expecting a ReactiveMongo document, found: $v")
73   }
