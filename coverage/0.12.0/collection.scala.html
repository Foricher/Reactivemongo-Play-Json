377   /**
378    * Inserts the document, or updates it if it already exists in the collection.
379    *
380    * @param doc The document to save.
381    */
382   @deprecated("0.11.1", "Use [[update]] with `upsert` set to true")
383   def save(doc: JsObject)(implicit ec: ExecutionContext): Future[WriteResult] =
384     save(doc, WriteConcern.Default)
385 
386   /**
387    * Inserts the document, or updates it if it already exists in the collection.
388    *
389    * @param doc The document to save.
390    * @param writeConcern The write concern
391    */
392   @deprecated("0.11.1", "Use [[update]] with `upsert` set to true")
393   def save(doc: pack.Document, writeConcern: WriteConcern)(implicit ec: ExecutionContext): Future[WriteResult] = {
394     import reactivemongo.bson.BSONObjectID
395     (doc \ "_id").toOption match {
396       case None => insert(
397         doc + ("_id" ->
398           BSONFormats.BSONObjectIDFormat.writes(BSONObjectID.generate)),
399         writeConcern
400       )
401 
402       case Some(id) =>
403         update(Json.obj("_id" -> id), doc, writeConcern, upsert = true)
404     }
405   }
406 
407   /**
408    * Inserts the document, or updates it if it already exists in the collection.
409    *
410    * @param doc The document to save.
411    * @param writeConcern The write concern
412    */
413   @deprecated("0.11.1", "Use [[update]] with `upsert` set to true")
414   def save[T](doc: T, writeConcern: WriteConcern = WriteConcern.Default)(implicit ec: ExecutionContext, writer: Writes[T]): Future[WriteResult] =
415     writer.writes(doc) match {
416       case d @ JsObject(_) => save(d, writeConcern)
417       case _ =>
418         Future.failed[WriteResult](new Exception("cannot write object"))
419     }

492 sealed trait JsCursor[T] extends Cursor[T] {
493   /**
494    * Returns the result of cursor as a JSON array.
495    *
496    * @param maxDocs Maximum number of documents to be retrieved
497    */
498   def jsArray(maxDocs: Int = Int.MaxValue)(implicit ec: ExecutionContext): Future[JsArray]
499 
500 }
501 
502 class JsCursorImpl[T: Writes](val wrappee: Cursor[T])
503     extends JsCursor[T] with WrappedCursor[T] {
504   import Cursor.{ Cont, Fail }
505 
506   private val writes = implicitly[Writes[T]]
507 
508   def jsArray(maxDocs: Int = Int.MaxValue)(implicit ec: ExecutionContext): Future[JsArray] = wrappee.foldWhile(Json.arr(), maxDocs)(
509     (arr, res) => Cont(arr :+ writes.writes(res)),
510     (_, error) => Fail(error)
511   )
512 
513 }
514 
515 class JsFlattenedCursor[T](val future: Future[JsCursor[T]])
516     extends FlattenedCursor[T](future) with JsCursor[T] {
517 
518   def jsArray(maxDocs: Int = Int.MaxValue)(implicit ec: ExecutionContext): Future[JsArray] = future.flatMap(_.jsArray(maxDocs))

520 }

549   /**
550    * Inserts the documents from a JSON source.
551    *
552    * @param documents the source that can be parsed as an array of JSON objects
553    * @param ordered true if to insert the document in order
554    * @param bulkSize the maximum size for each insert bulk
555    * @param wc the write concern
556    */
557   def bulkInsert(collection: JSONCollection, documents: => InputStream, ordered: Boolean, bulkSize: Int, bulkByteSize: Int)(implicit ec: ExecutionContext, wc: WriteConcern): Future[MultiBulkWriteResult] =
558     documentProducer(collection, documents).flatMap { producer =>
559       collection.bulkInsert(ordered, wc, bulkSize, bulkByteSize)(producer: _*)
560     }
561 
562   /**
563    * Inserts the documents from a JSON source.
564    *
565    * @param documents the source that can be parsed as an array of JSON objects
566    * @param ordered true if to insert the document in order
567    * @param wc the write concern
568    */
569   def bulkInsert(collection: JSONCollection, documents: => InputStream, ordered: Boolean = true)(implicit ec: ExecutionContext, wc: WriteConcern = collection.db.connection.options.writeConcern): Future[MultiBulkWriteResult] =
570     documentProducer(collection, documents).flatMap { producer =>
571       collection.bulkInsert(ordered, wc)(producer: _*)
572     }
573 
574   private def documentProducer(collection: JSONCollection, documents: => InputStream)(implicit ec: ExecutionContext): Future[List[collection.ImplicitlyDocumentProducer]] = {
575     lazy val in = documents
576     def docs: Future[List[JsObject]] = try {
577       Json.parse(in).validate[List[JsObject]] match {
578         case JsSuccess(os, _) => Future.successful(os)
579         case err @ JsError(_) => Future.failed(new Exception(
580           Json.stringify(JsError toJson err)
581         ))
582       }
583     } catch {
584       case reason: Throwable => Future.failed(reason)
585     } finally {
586       try {
587         in.close()
588       } catch {
589         case _: Throwable => ()
590       }
591     }
592 
593     docs.map {
594       _.map { implicitly[collection.ImplicitlyDocumentProducer](_) }
595     }
596   }
